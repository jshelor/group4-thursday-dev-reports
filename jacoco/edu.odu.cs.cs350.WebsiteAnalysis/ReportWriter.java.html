package edu.odu.cs.cs350.WebsiteAnalysis;
import java.sql.Timestamp;
import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;

import javax.management.RuntimeErrorException;

public class ReportWriter {
    public ReportWriter(Website website, String fileType)
    {
        this.website = website;
        this.fileType = fileType;
        //012345678901234567890123456789
        //yyyy-mm-dd hh:mm:ss.fffffffff
        Timestamp timestamp = new Timestamp(System.currentTimeMillis());
        timeStamp = timestamp.toString();
        timeStamp = timeStamp.substring(0, 3)+timeStamp.substring(5, 6)+timeStamp.substring(8, 9)+"-"+timeStamp.substring(11, 12)+timeStamp.substring(14, 15)+timeStamp.substring(17, 18);
    }
    public ArrayList<String> generateContent()
    {
        return null;  
    }
    public void writeReport(ArrayList<String> content)
    {
        String fileName = timeStamp + "-summary." + fileType;
        try 
        {
            FileWriter writer = new FileWriter(fileName);
            for(String line: content)
            {
                writer.write(line + "\n");   
            }
            writer.close();
        }
        catch(IOException e)
        {

        }
    }
    protected Website website;
    private String fileType;
    private String timeStamp;
}
