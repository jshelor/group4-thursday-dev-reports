package edu.odu.cs.cs350.WebsiteAnalysis;

public class Image extends Resource {
    private float megabytes;
    /**
     * Initialize a new Image data object with the URI, location type, and file size.
     * @param URI The URI of the image.
     * @param locationType The scope relative to the site root.
     * @param fileSize The file size in MiB.
     */
    Image(String URI, Classification locationType, float megabytes) {
        super(URI, locationType);
        this.megabytes = megabytes;
    }

    /**
     * Retrieve the file size of the image in megabytes.
     * @return The file size in binary megabytes.
     */
    public final float fileSize() {
        return this.megabytes;
    }
}
