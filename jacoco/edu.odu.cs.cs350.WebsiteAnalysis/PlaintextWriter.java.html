package edu.odu.cs.cs350.WebsiteAnalysis;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Comparator;
import java.util.Collections;

public class PlaintextWriter extends ReportWriter {
  public PlaintextWriter(Website website) {
    super(website, "txt");
  }

  public ArrayList<String> generateContent()
    {
        ArrayList<HTMLDocument> documents = (ArrayList<HTMLDocument>) website.getHTMLDocuments();
        HTMLDocument[] docsAsArray = new HTMLDocument[documents.size()];
        docsAsArray = documents.toArray(docsAsArray);
        Comparator<HTMLDocument> byPath = Comparator.comparing(HTMLDocument::getPath);
        Arrays.sort(docsAsArray, byPath);
        
        ArrayList<String> content = new ArrayList<String>();
       // double totalSizeMB = 0.0;
        for(HTMLDocument document: docsAsArray)
        {
            //need images to be itterable 
           // double sizeMB = document.getSizeOfImages();
           // content.add(String.format("%7.2f %s\n", sizeMB, document.getPath()));
            //totalSizeMB += sizeMB;
            content.add(document.getPath());
        }
        //content.add(String.format("%7.2f Total (MBs)\n", totalSizeMB));
        //super.writeReport(content);
        return content;
    }
    
}
