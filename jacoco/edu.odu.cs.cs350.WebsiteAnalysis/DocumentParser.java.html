package edu.odu.cs.cs350.WebsiteAnalysis;

import java.util.List;
import java.util.ArrayList;

import org.jsoup.nodes.*;



public class DocumentParser {
    private String absolutePath;

    /**
     * Instantiates a new DocumentParser to extract information about a locally-saved HTML file.
     * @param absolutePath The full local file path to a file containing HTML content for parsing.
     */
    DocumentParser(String absolutePath) {
        this.absolutePath = absolutePath;
    }
    
    /**
     * Direct DocumentParser to analyze the local page's links.
     * @return A List of Anchor objects containing the information collected.
     */
    public List<Anchor> parseLinks() {
        ArrayList<Anchor> theLinks = new ArrayList<>();

        return theLinks;
    }

    /**
     * Direct DocumentParser to analyze the page's images.
     * @return A list of Image objects containing the information collected.
     */
    public List<Image> parseImages() {
        ArrayList<Image> theImages = new ArrayList<>();

        return theImages;
    }

    /**
     * Direct DocumentParser to analyze the page's JavaScript links.
     * @return A list of Script objects containing the information collected.
     */
    public List<Script> parseJS() {
        ArrayList<Script> theJS = new ArrayList<>();

        return theJS;    
    }

    /**
     * Direct DocumentParser to analyze the page's stylesheets.
     * @return A list of Stylesheet objects containing the information collected.
     */
    public List<Stylesheet> parseCSS() {
        ArrayList<Stylesheet> theCSS = new ArrayList<>();

        return theCSS;
    }

}
