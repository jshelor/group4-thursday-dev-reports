package edu.odu.cs.cs350.WebsiteAnalysis;

public abstract class Resource {
    /**
     * The URI to the resource. Should a String that can be parsed as a URI.
     */
    private String URI;
    private Classification locationType;

    /**
     * Initialize a new Resource object with the URI and Classification.
     * @param URI The URI, given as a String.
     * @param locationType The location type, external, internal, or intrapage.
     */
    Resource(String URI, Classification locationType) {
        this.URI = URI;
        this.locationType = locationType;
    }

    /**
     * Get the page resource's URI.
     * @return A URI in String format.
     */
    public String getURI() {
        return URI;
    }

    /**
     * Get the page resource's classification.
     * @return The scope of the resource relative to the site root.
     */
    public Classification getLocationType() {
        return locationType;
    }

}
